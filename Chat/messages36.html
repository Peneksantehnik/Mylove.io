public class TodoListController {

this.todoList = todoList;
this.todoListView = todoListView;
}

public void processInput(String input) {
switch (input) {
case "add":
todoList.addTask(todoListView.getTask());
break;
case "remove":
todoList.removeTask(todoListView.getIndex() - 1);
break;
case "print":
todoListView.printTasks(todoList.getTasks());
break;
case "quit":
System.exit(0);
break;
default:
System.out.println("Invalid command");
break;
}
}
